title Staff Reset Password (Class-based Sequence Diagram with Next.js Frontend)

actor "Administrative Staff" as Staff
participant "resetPage : StaffPasswordResetPage" as Page
participant "resetView : StaffPasswordResetView" as View
participant "authApiClient : AuthApiClient" as ApiClient
participant "authCtrl : AuthController" as Controller
participant "authAppService : AuthApplicationService" as AppService
participant "authDomainService : AuthDomainService" as DomainService
participant "tokenRepo : PasswordResetTokenRepository" as TokenRepo
participant "userRepo : UserRepository" as UserRepo
participant "emailService : EmailNotificationService" as EmailService
database "db : PostgreSQL" as DB

Staff -> Page: navigateToStaffResetPassword()
activate Page
Page -> View: renderResetForm()
activate View
View --> Staff: display reset form

Staff -> View: submitEmail(email)
View -> ApiClient: postForgotPassword(email)
activate ApiClient
ApiClient -> Controller: POST /auth/staff/forgot-password
activate Controller
Controller -> AppService: requestStaffPasswordReset(email)
activate AppService
AppService -> DomainService: createResetTokenIfStaffExists(email)
activate DomainService
DomainService -> UserRepo: findByEmail(email)
activate UserRepo
UserRepo -> DB: query staff user
activate DB
DB --> UserRepo: user / null
deactivate DB
UserRepo --> DomainService: user / null
deactivate UserRepo

alt [User Exists]
    DomainService -> TokenRepo: save reset token
    activate TokenRepo
    TokenRepo -> DB: insert token
    activate DB
    DB --> TokenRepo: insert success
    deactivate DB
    TokenRepo --> DomainService: token saved
    deactivate TokenRepo
    DomainService -> EmailService: sendResetLink(email)
    activate EmailService
    EmailService --> DomainService: mail sent
    deactivate EmailService
end

DomainService --> AppService: generic success response
deactivate DomainService
AppService --> Controller: response
deactivate AppService
Controller --> ApiClient: HTTP 200 OK
deactivate Controller
ApiClient --> View: return success
deactivate ApiClient
View --> Staff: display generic reset message

deactivate View
deactivate Page
