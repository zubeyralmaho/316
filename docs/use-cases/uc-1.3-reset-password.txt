@startuml
title UC-1.3 Reset Password (Class-based Sequence Diagram with Next.js Frontend)

actor "Applicant Student" as Student
participant "resetPage : PasswordResetPage" as ResetPage
participant "resetView : PasswordResetView" as ResetView
participant "authApiClient : AuthApiClient" as ApiClient
participant "authCtrl : AuthController" as Controller
participant "authAppService : AuthApplicationService" as AppService
participant "authDomainService : AuthDomainService" as DomainService
participant "tokenRepo : PasswordResetTokenRepository" as TokenRepo
participant "userRepo : UserRepository" as UserRepo
participant "emailService : EmailNotificationService" as EmailService
database "db : PostgreSQL" as DB
participant "loginPage : LoginPage" as LoginPage

== Step 1: Open Reset Password Page ==
Student -> ResetPage: 1. navigateToResetPassword()
activate ResetPage
ResetPage -> ResetView: 1.1 renderResetRequestForm()
activate ResetView
ResetView --> Student: 1.2 display reset request form
deactivate ResetView
deactivate ResetPage

== Step 2: Submit Reset Request ==
Student -> ResetView: 2. submitEmail(email)
activate ResetView

alt [UC-1.3.EX.1 Invalid Input]
    ResetView --> Student: 2.1 display "Please enter a valid email address"

else [Valid Input]
    ResetView -> ApiClient: 2.2 postForgotPassword(email)
    activate ApiClient

    ApiClient -> Controller: 2.3 POST /auth/forgot-password
    activate Controller

    Controller -> AppService: 2.4 requestPasswordReset(email)
    activate AppService

    AppService -> DomainService: 2.5 createResetTokenIfUserExists(email)
    activate DomainService

    DomainService -> UserRepo: 2.6 findByEmail(email)
    activate UserRepo
    UserRepo -> DB: 2.7 query user by email
    activate DB
    DB --> UserRepo: 2.8 user / null
    deactivate DB
    UserRepo --> DomainService: 2.9 user / null
    deactivate UserRepo

    alt [UC-1.3.EX.2 Email Not Found]
        DomainService --> AppService: 2.10 generic success response
        AppService --> Controller: 2.11 generic response
        Controller --> ApiClient: 2.12 HTTP 200 OK
        ApiClient --> ResetView: 2.13 return generic success
        ResetView --> Student: 2.14 display "If an account exists, a reset link has been sent"

    else [User Exists]
        DomainService -> TokenRepo: 2.15 save reset token
        activate TokenRepo
        TokenRepo -> DB: 2.16 insert token
        activate DB
        DB --> TokenRepo: 2.17 insert success
        deactivate DB
        TokenRepo --> DomainService: 2.18 token saved
        deactivate TokenRepo

        DomainService -> EmailService: 2.19 sendResetLink(email, resetLink)
        activate EmailService
        EmailService --> DomainService: 2.20 email queued/sent
        deactivate EmailService

        DomainService --> AppService: 2.21 generic success response
        AppService --> Controller: 2.22 generic response
        Controller --> ApiClient: 2.23 HTTP 200 OK
        ApiClient --> ResetView: 2.24 return success
        ResetView --> Student: 2.25 display "If an account exists, a reset link has been sent"
    end

    deactivate DomainService
    deactivate AppService
    deactivate Controller
    deactivate ApiClient
    deactivate ResetView
end

== Step 3: Open Reset Link ==
Student -> ResetPage: 3. openResetLink(token)
activate ResetPage
ResetPage -> ResetView: 3.1 renderSetNewPasswordForm()
activate ResetView
ResetView --> Student: 3.2 display new password form

== Step 4: Submit New Password ==
Student -> ResetView: 4. submitNewPassword(password, confirmPassword, token)

alt [UC-1.3.EX.3 Password Policy Violation]
    ResetView --> Student: 4.1 display password policy errors

else [Passwords valid]
    ResetView -> ApiClient: 4.2 postResetPassword(token, newPassword)
    activate ApiClient

    ApiClient -> Controller: 4.3 POST /auth/reset-password
    activate Controller

    Controller -> AppService: 4.4 resetPassword(token, newPassword)
    activate AppService

    AppService -> DomainService: 4.5 validateTokenAndUpdatePassword(token, newPassword)
    activate DomainService

    DomainService -> TokenRepo: 4.6 findValidToken(token)
    activate TokenRepo
    TokenRepo -> DB: 4.7 query token
    activate DB
    DB --> TokenRepo: 4.8 token / null / expired
    deactivate DB
    TokenRepo --> DomainService: 4.9 token result
    deactivate TokenRepo

    alt [UC-1.3.EX.4 Invalid or Expired Token]
        DomainService --> AppService: 4.10 throw InvalidResetTokenException
        AppService --> Controller: 4.11 error response
        Controller --> ApiClient: 4.12 HTTP 400 Bad Request
        ApiClient --> ResetView: 4.13 return error
        ResetView --> Student: 4.14 display "This reset link is invalid or has expired"

    else [Normal Course]
        DomainService -> UserRepo: 4.15 update password hash
        activate UserRepo
        UserRepo -> DB: 4.16 update user password
        activate DB
        DB --> UserRepo: 4.17 update success
        deactivate DB
        UserRepo --> DomainService: 4.18 user updated
        deactivate UserRepo

        DomainService -> TokenRepo: 4.19 invalidate token
        activate TokenRepo
        TokenRepo -> DB: 4.20 update token status
        activate DB
        DB --> TokenRepo: 4.21 update success
        deactivate DB
        TokenRepo --> DomainService: 4.22 token invalidated
        deactivate TokenRepo

        DomainService --> AppService: 4.23 reset success
        deactivate DomainService

        AppService --> Controller: 4.24 success response
        deactivate AppService

        Controller --> ApiClient: 4.25 HTTP 200 OK
        deactivate Controller

        ApiClient --> ResetView: 4.26 return success
        deactivate ApiClient

        ResetView -> LoginPage: 4.27 redirectToLogin()
        activate LoginPage
        LoginPage --> Student: 4.28 display login page with success message
        deactivate LoginPage
    end
end

deactivate ResetView
deactivate ResetPage
@enduml
