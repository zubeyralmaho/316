@startuml
title Review Applicant Documents (Generic Staff Sequence Diagram with Next.js Frontend)

actor "Administrative Staff" as Staff
participant "reviewPage : ApplicantReviewPage" as Page
participant "reviewView : ApplicantReviewView" as View
participant "documentApiClient : DocumentApiClient" as ApiClient
participant "documentCtrl : DocumentController" as Controller
participant "documentAppService : DocumentApplicationService" as AppService
participant "documentDomainService : DocumentDomainService" as DomainService
participant "applicationRepo : ApplicationRepository" as AppRepo
participant "documentRepo : DocumentRepository" as DocumentRepo
participant "storageService : StorageService" as StorageService
participant "statusService : StatusApplicationService" as StatusService
participant "notificationService : NotificationApplicationService" as NotificationService
database "db : PostgreSQL" as DB
collections "minio : MinIO Object Storage" as MinIO

== Step 1: Open Applicant Review Page ==
Staff -> Page: 1. navigateToApplicantReview(applicationId)
activate Page
Page -> View: 1.1 renderApplicantDocumentsPage()
activate View

View -> ApiClient: 1.2 getApplicantDocuments(applicationId)
activate ApiClient
ApiClient -> Controller: 1.3 GET /documents/application/{applicationId}
activate Controller
Controller -> AppService: 1.4 getApplicationDocuments(applicationId)
activate AppService
AppService -> AppRepo: 1.5 findById(applicationId)
activate AppRepo
AppRepo -> DB: 1.6 query application
activate DB
DB --> AppRepo: 1.7 application found
deactivate DB
AppRepo --> AppService: 1.8 application
deactivate AppRepo

AppService -> DocumentRepo: 1.9 findAllByApplicationId(applicationId)
activate DocumentRepo
DocumentRepo -> DB: 1.10 query documents
activate DB
DB --> DocumentRepo: 1.11 document metadata list
deactivate DB
DocumentRepo --> AppService: 1.12 documents
deactivate DocumentRepo

AppService --> Controller: 1.13 DocumentListResponse
deactivate AppService
Controller --> ApiClient: 1.14 HTTP 200 OK
deactivate Controller
ApiClient --> View: 1.15 return documents
deactivate ApiClient
View --> Staff: 1.16 display uploaded documents

== Step 2: Open a Specific Document ==
Staff -> View: 2. selectDocument(documentId)
View -> ApiClient: 2.1 getDocumentPreview(documentId)
activate ApiClient
ApiClient -> Controller: 2.2 GET /documents/{documentId}
activate Controller
Controller -> AppService: 2.3 getDocument(documentId)
activate AppService
AppService -> DocumentRepo: 2.4 findById(documentId)
activate DocumentRepo
DocumentRepo -> DB: 2.5 query document
activate DB
DB --> DocumentRepo: 2.6 document metadata
deactivate DB
DocumentRepo --> AppService: 2.7 document metadata
deactivate DocumentRepo

AppService -> StorageService: 2.8 getDocument(document.storageKey)
activate StorageService
StorageService -> MinIO: 2.9 fetch object
activate MinIO
MinIO --> StorageService: 2.10 file stream / not found
deactivate MinIO

alt [EX.1 File Corrupted / Cannot Open]
    StorageService --> AppService: 2.11 throw FileAccessException
    AppService --> Controller: 2.12 error response
    Controller --> ApiClient: 2.13 HTTP 422 Unprocessable Entity
    ApiClient --> View: 2.14 return file preview error
    View --> Staff: 2.15 display "File cannot be viewed"

else [Normal Course]
    StorageService --> AppService: 2.16 file stream
    deactivate StorageService
    AppService --> Controller: 2.17 DocumentPreviewResponse
    deactivate AppService
    Controller --> ApiClient: 2.18 HTTP 200 OK
    deactivate Controller
    ApiClient --> View: 2.19 return preview
    deactivate ApiClient
    View --> Staff: 2.20 display document preview
end

== Step 3: Review Decision ==
alt [AC.1 Approve Document]
    Staff -> View: 3. approveDocument(documentId)
    View -> ApiClient: 3.1 patchReviewStatus(documentId, APPROVED)
    activate ApiClient
    ApiClient -> Controller: 3.2 PATCH /documents/{documentId}/review
    activate Controller
    Controller -> AppService: 3.3 reviewDocument(documentId, APPROVED, null)
    activate AppService
    AppService -> DomainService: 3.4 reviewDocument(documentId, APPROVED)
    activate DomainService
    DomainService -> DocumentRepo: 3.5 updateReviewStatus(documentId, APPROVED)
    activate DocumentRepo
    DocumentRepo -> DB: 3.6 update document status
    activate DB
    DB --> DocumentRepo: 3.7 update success
    deactivate DB
    DocumentRepo --> DomainService: 3.8 updated
    deactivate DocumentRepo
    DomainService --> AppService: 3.9 review success
    deactivate DomainService
    AppService --> Controller: 3.10 success response
    deactivate AppService
    Controller --> ApiClient: 3.11 HTTP 200 OK
    deactivate Controller
    ApiClient --> View: 3.12 return success
    deactivate ApiClient
    View --> Staff: 3.13 display "Document approved"

else [AC.2 Reject Document]
    Staff -> View: 3.14 rejectDocument(documentId, reason)
    View -> ApiClient: 3.15 patchReviewStatus(documentId, REJECTED, reason)
    activate ApiClient
    ApiClient -> Controller: 3.16 PATCH /documents/{documentId}/review
    activate Controller
    Controller -> AppService: 3.17 reviewDocument(documentId, REJECTED, reason)
    activate AppService
    AppService -> DomainService: 3.18 reviewDocument(documentId, REJECTED)
    activate DomainService
    DomainService -> DocumentRepo: 3.19 updateReviewStatus(documentId, REJECTED)
    activate DocumentRepo
    DocumentRepo -> DB: 3.20 update document status
    activate DB
    DB --> DocumentRepo: 3.21 update success
    deactivate DB
    DocumentRepo --> DomainService: 3.22 updated
    deactivate DocumentRepo
    DomainService --> AppService: 3.23 rejected
    deactivate DomainService

    AppService -> StatusService: 3.24 updateStatus(applicationId, UNDER_INITIAL_REVIEW)
    activate StatusService
    StatusService -> DB: 3.25 update status/history
    activate DB
    DB --> StatusService: 3.26 update success
    deactivate DB
    StatusService --> AppService: 3.27 status updated
    deactivate StatusService

    AppService -> NotificationService: 3.28 createActionRequiredNotification(applicationId, reason)
    activate NotificationService
    NotificationService -> DB: 3.29 insert notification
    activate DB
    DB --> NotificationService: 3.30 insert success
    deactivate DB
    NotificationService --> AppService: 3.31 notification created
    deactivate NotificationService

    AppService --> Controller: 3.32 success response
    deactivate AppService
    Controller --> ApiClient: 3.33 HTTP 200 OK
    deactivate Controller
    ApiClient --> View: 3.34 return success
    deactivate ApiClient
    View --> Staff: 3.35 display "Document rejected and applicant notified"

else [AC.3 Request Additional Document]
    Staff -> View: 3.36 requestAdditionalDocument(applicationId, message)
    View -> ApiClient: 3.37 postAdditionalDocumentRequest(applicationId, message)
    activate ApiClient
    ApiClient -> Controller: 3.38 POST /documents/request-additional
    activate Controller
    Controller -> AppService: 3.39 requestAdditionalDocument(applicationId, message)
    activate AppService

    AppService -> StatusService: 3.40 updateStatus(applicationId, UNDER_INITIAL_REVIEW)
    activate StatusService
    StatusService -> DB: 3.41 update status/history
    activate DB
    DB --> StatusService: 3.42 update success
    deactivate DB
    StatusService --> AppService: 3.43 status updated
    deactivate StatusService

    AppService -> NotificationService: 3.44 createActionRequiredNotification(applicationId, message)
    activate NotificationService
    NotificationService -> DB: 3.45 insert notification
    activate DB
    DB --> NotificationService: 3.46 insert success
    deactivate DB
    NotificationService --> AppService: 3.47 notification created
    deactivate NotificationService

    AppService --> Controller: 3.48 success response
    deactivate AppService
    Controller --> ApiClient: 3.49 HTTP 200 OK
    deactivate Controller
    ApiClient --> View: 3.50 return success
    deactivate ApiClient
    View --> Staff: 3.51 display "Additional document request sent"
end

== Exception Flow ==
alt [EX.2 Document Repository Unavailable]
    Controller --> ApiClient: E1 HTTP 503 Service Unavailable
    ApiClient --> View: E2 return repository error
    View --> Staff: E3 display "Documents are temporarily unavailable"
end

deactivate View
deactivate Page
@enduml
